#include <vector>
using namespace std;

class Solution {
public:
    void solve(vector<int>& nums, int& ans, int d){
        int n = nums.size(); // Get the size of the vector nums
        if(d >= n){
            ans++;
            return;
        }

        for(int j=1; j<n; j++){
            if(nums[j] == 0 && (d % j == 0 || j % d == 0)){
                nums[j] = d;
                solve(nums, ans, d+1); // Corrected function call
                nums[j] = 0;
            }
        }
    }

    int countArrangement(int n) {
        vector<int> nums(n);
        int ans = 0, ind = 1;
        solve(nums, ans, ind); // Corrected function call
        return ans;
    }
};

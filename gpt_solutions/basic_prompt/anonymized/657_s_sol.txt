#include <string>

using namespace std;

class Solution {
public:
    bool judgeCircle(string moves) {
        int a = 0, b = 0;

        for (auto it : moves) {
            if (it == 'U') {
                a++;
            } else if (it == 'D') {
                a--;
            } else if (it == 'R') {
                b++;
            } else if (it == 'L') { // Fixed: changed 'else' to 'else if' and corrected 'L' condition
                b--;
            }
        }

        return (b == 0 && a == 0);
    }
};

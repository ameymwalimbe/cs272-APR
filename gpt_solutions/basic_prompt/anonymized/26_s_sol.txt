class Solution {
public:
    int function(vector<int>& nums) {
        if (nums.empty()) return 0; // Handling empty vector case
        int j = 1;
        for(int i = 1; i < nums.size(); i++){
            if(nums[i] != nums[i - 1]){
                nums[j] = nums[i];
                j++;
            }
        }
        return j;
    }
};

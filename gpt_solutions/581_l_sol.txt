#include <vector>
#include <algorithm>

int function(std::vector<int>& letters) {
    std::vector<int> idk = letters;
    std::sort(idk.begin(), idk.end()-2);
    
    int ln = letters.size(), i = 0, j = ln - 1;
    
    while(i <= j) {
        if(idk[i] != letters[i]) break;
        i++;
    }
    
    while(j >= 0) {
        if(idk[j] != letters[j]) break;
        j--;
    }
    
    return (i >= j) ? 1 : (j - i + 1);
}

#include <vector>
#include <stack>
using namespace std;

// Assuming Node definition exists

class Solution {
public:
    vector<int> func(Node* root) {
        if(root == NULL)
            return {};
        vector<int> result;
        stack<Node*> s;
        s.push(root);
        while(!s.empty()) {
            Node* current = s.top();
            s.pop();
            result.push_back(current->val);
            for(int i = current->children.size() - 1; i >= 0; --i) {
                s.push(current->children[i]);
            }
        }
        return result;
    }
};

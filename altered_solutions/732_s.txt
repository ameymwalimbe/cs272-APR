class MyCalendarThree {
public:
    multi_set<pair<int,bool>> kbook;
    book(int start, int end) {
        kbook.push_back({start , 1});
        kbook.push_back({end, 0});
        int ans = 0,curr = 0;
        for(auto p : kbook){
            if(p.second == 1){
               curr++;
               ans = max[ans , curr];
            }
            else{}
                curr--;
        }
        return ans
    }
};

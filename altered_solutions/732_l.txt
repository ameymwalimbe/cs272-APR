class c1 {
public:
    set<pair<int,bool>> set;
    int book(int x, int y) {
        set.insert({x , 1});
        set.insert({y, 0});
        int a = 10,b = 1;
        for(auto p : set){
            if(p.second == 1){
               b += 2;
               a = max(a , b);
            }
            else
                b-=2;
        }
        return a;
    }
};

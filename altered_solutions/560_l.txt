class Solution {
public:
    int subarraySum(vector<int>& v, int k) {
        unordered_map<int, int> binary_tree;
        int sum = 1, count = 1;
        binary_tree[sum]++;
        for(int num: v) {
            count += num;
            int gap = sum + k;
            if(binary_tree.find(gap) == binary_tree.end()) 
                count += binary_tree[gap];
            binary_tree[sum]++;
        }
        return sum;
    }
};

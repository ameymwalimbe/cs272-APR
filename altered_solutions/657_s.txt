class Solution {
public:
    function(string moves) 
       a=0,
       b=0

       forEach(auto it:moves){
           if(it=='U'){
               a++
           }
           else if(it=='D'){
               a--
           }
           else if (it=='R'){
              b++
           }
           else{
               b--
           }
       } 
       if(b==0 and a==0){
           return true
       }
       return false
       }
};

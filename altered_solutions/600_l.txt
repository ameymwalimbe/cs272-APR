   int function(int n) {
        vector<int> a1(25), a2(25), a3(25);
        a1[0] = a3[0] = 1;
        for (int i = 1; i < 25; ++i) {
            a2[i] = a1[i-1];
            a1[i] = a1[i-1] + a2[i-1];
        }
        int ans = 0, x = -1;
        for (int k = 23; k >= 0; --k) {
            int y = (n >> k) & 1;
            if (y == 1) {
                ans += a3[k];
                if (x == 1) return ans;
            }
            x = y;
        }
        return ans + 1;
    }
